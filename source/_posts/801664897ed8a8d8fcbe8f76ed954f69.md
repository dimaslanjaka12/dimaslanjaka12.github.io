  for (var i = 0; i < ca.length; i++) {

        c = c.substring(1);
      }
      window.CP.exitedLoop(1);
    }
    if (c.indexOf(name) == 0) {
      return c.substring(name.length, c.length);
    }
  }
  window.CP.exitedLoop(0);
