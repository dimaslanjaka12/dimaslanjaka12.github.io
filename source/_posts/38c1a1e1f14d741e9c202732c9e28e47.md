var thisIsObject= {
    'Cow' : 'Moo',
    'Cat' : 'Meow',
    'Dog' : 'Bark'
};
try {
  delete thisIsObject['Cow'];
} catch(e){
  thisIsObject.cow = undefined;
}//test using developer tools F12
console.log(thisIsObject);
 Output 
=> {Cat: "Meow", Dog: "Bark"}
 Wrapping in function for Easy Use 
function delkey(obj, key){
  try {
    delete obj[key];
  } catch(e){
    obj[key] = undefined;
  }
  return obj;
}
