var oddoreven = function (n = false, type = 'odd') {
  var time = !n ? new Date().getDay() : Number(n);
  
  if (!/^-{0,1}\d+$/.test(time)){
    alert('arguments is not number, please remove quote');
    return;
  }
  hasil = time % 2;
  return hasil == (type == ('odd' || 'ganjil') ? 1 : 0);
}
