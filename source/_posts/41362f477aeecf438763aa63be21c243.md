  var ca = decodedCookie.split(';');
  for (var i = 0; i < ca.length; i++) {
    if (window.CP.shouldStopExecution(0)) break;
    var c = ca[i];
    while (c.charAt(0) == ' ') {

      c = c.substring(1);
    }
    window.CP.exitedLoop(1);
    if (c.indexOf(name) == 0) {

    }
  }
  window.CP.exitedLoop(0);

}
